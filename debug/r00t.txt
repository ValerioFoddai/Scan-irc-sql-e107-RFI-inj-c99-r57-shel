sub links() { 
    my @l; 
    my $link = $_[0]; 
    my $host = $_[0]; 
    my $hdir = $_[0]; 
    $hdir =~ s/(.*)\/[^\/]*$/\1/; 
    $host =~ s/([-a-zA-Z0-9\.]+)\/.*/$1/; 
    $host .= "/"; 
    $link .= "/"; 
    $hdir .= "/"; 
    $host =~ s/\/\//\//g; 
    $hdir =~ s/\/\//\//g; 
    $link =~ s/\/\//\//g; 
    push( @l, $link, $host, $hdir ); 
    return @l; 
}

sub key() { 
    my $dork = $_[0];

    $dork =~ s/:/\%3A/g; 
    $dork =~ s/\//\%2F/g; 
    $dork =~ s/&/\%26/g; 
    $dork =~ s/\"/\%22/g; 
    $dork =~ s/,/\%2C/g; 
    $dork =~ s/\\/\%5C/g; 
    return $dork; 
} 

sub end() { 
    $string = $_[0]; 
    $string .= "/"; 
    $string =~ s/\/\//\//; 
    while ( $string =~ /\/\// ) { 
        $string =~ s/\/\//\//; 
    } 
    return ($string); 
} 

sub query($) { 
    my $url = $_[0]; 
    $url =~ s/http:\/\///; 
    my $host  = $url; 
    my $query = $url; 
    my $page  = ""; 
    $host  =~ s/href=\"?http:\/\///; 
    $host  =~ s/([-a-zA-Z0-9\.]+)\/.*/$1/; 
    $query =~ s/$host//; 
    if ( $query eq "" ) { $query = "/"; } 
    eval { 
        my $sock = IO::Socket::INET->new( 
            PeerAddr => "$host", 
            PeerPort => "80", 
            Proto    => "tcp" 
        ) or return; 
        print $sock 
"GET $query HTTP/1.0\r\nHost: $host\r\nAccept: */*\r\nUser-Agent: Mozilla/5.0\r\n\r\n"; 
        my @r = <$sock>; 
        $page = "@r"; 
        close($sock); 
    }; 
    return $page; 
} 

sub get_link() { 
    my $file_print = $_[1]; 
    my $link       = $_[0]; 
    my $host       = $_[0]; 
    my $host_dir   = $_[0]; 
    my @links; 
    $host_dir =~ s/(.*)\/[^\/]*$/\1/; 
    $host     =~ s/([-a-zA-Z0-9\.]+)\/.*/$1/; 
    $host_dir = &end($host_dir); 
    $host     = &end($host); 
    $link     = &end($host); 
    push( @links, $link, $host, $host_dir ); 
    open( $file, '>>', $file_print ); 
    print $file "$link\n$host_dir\n$host\n"; 
    close($file); 
    return @links; 
} 

sub unici { 
    my @unici = (); 
    my %visti = (); 
    foreach my $elemento (@_) { 
        $elemento =~ s/\/+/\//g; 
        next if $visti{$elemento}++; 
        push @unici, $elemento; 
    } 
    return @unici; 
}
